<script>

const supabase = window.supabase.createClient(
  "SUPABASE_URL",
  "SUPABASE_KEY"
);

let userId = "";

// 🔐 pobranie usera
async function loadUser() {
  const session = await supabase.auth.getSession();
  userId = session.data.session.user.id;

  loadData();
}

// 📊 pobranie danych
async function loadData() {
  const { data } = await supabase
    .from("users")
    .select("*")
    .eq("id", userId)
    .single();

  document.getElementById("plan").innerText = data.plan;

  document.getElementById("credits").innerText =
    data.plan === "pro" ? "Nielimitowane 🔥" : data.credits;

  document.getElementById("expiry").innerText =
    data.plan_expiry || "Brak";

  document.getElementById("stats").innerText =
    "Publikacje: " + data.published_count +
    " | Dziennie: " + data.daily_count +
    " | Miesięcznie: " + data.monthly_count;
}

// 💸 PRO
function buyPro() {
  window.location.href = "LINK_STRIPE";
}

// 🚪 logout
async function logout() {
  await supabase.auth.signOut();
  window.location.href = "/";
}

loadUser();

</script>
