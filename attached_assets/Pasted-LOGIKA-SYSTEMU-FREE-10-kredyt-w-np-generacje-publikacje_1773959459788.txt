// --- SPRAWDZANIE KREDYTÓW ---
async function checkCredits(userId) {
  const { data } = await supabase
    .from("users")
    .select("*")
    .eq("id", userId)
    .single();

  const now = new Date();

  // jeśli PRO i aktywny
  if (data.plan === "pro" && new Date(data.plan_expiry) > now) {
    return true;
  }

  // FREE
  if (data.credits > 0) return true;

  return false;
}
3️⃣ Odejmowanie kredytów
async function useCredit(userId) {
  await supabase
    .from("users")
    .update({ credits: supabase.raw("credits - 1") })
    .eq("id", userId);
}
4️⃣ Reset kredytów co miesiąc
async function resetCreditsIfNeeded(userId) {
  const { data } = await supabase
    .from("users")
    .select("*")
    .eq("id", userId)
    .single();

  const now = new Date();

  // jeśli PRO wygasło → wraca do FREE
  if (data.plan === "pro" && new Date(data.plan_expiry) < now) {
    await supabase.from("users").update({
      plan: "free",
      credits: 10
    }).eq("id", userId);
  }
}

app.post("/generate", async (req, res) => {
  const { userId, prompt } = req.body;

  await resetCreditsIfNeeded(userId);

  const allowed = await checkCredits(userId);
  if (!allowed) return res.json({ error: "Brak kredytów 💸" });

  // AI GENEROWANIE
  const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.GROQ_API_KEY}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      model: "llama3-70b-8192",
      messages: [{ role: "user", content: prompt }]
    })
  });

  const data = await response.json();
  const html = data.choices[0].message.content;

  // odejmij kredyt jeśli FREE
  await useCredit(userId);

  res.json({ html });
