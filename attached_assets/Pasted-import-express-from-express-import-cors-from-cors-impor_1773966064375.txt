import express from "express";
import cors from "cors";
import fetch from "node-fetch";
import Stripe from "stripe";
import { createClient } from "@supabase/supabase-js";

const app = express();
app.use(cors());
app.use(express.json());

// ------------------- CONFIG -------------------
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_ANON_KEY
);

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

// ------------------- AI AGENT -------------------
app.post("/agent", async (req, res) => {
  let { html } = req.body;

  for (let i = 0; i < 3; i++) {
    // Analiza
    const analysisRes = await fetch("https://api.groq.com/openai/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.GROQ_API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "llama3-70b-8192",
        messages: [
          { role: "system", content: "Analizujesz kod HTML/CSS/JS i znajdujesz błędy." },
          { role: "user", content: html }
        ]
      })
    });
    const analysisData = await analysisRes.json();
    const issues = analysisData.choices[0].message.content;

    // Poprawka
    const fixRes = await fetch("https://api.groq.com/openai/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.GROQ_API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "llama3-70b-8192",
        messages: [
          { role: "system", content: "Jesteś ekspertem frontend. Popraw kod HTML/CSS/JS zgodnie z problemami." },
          { role: "user", content: `Kod:\n${html}\nProblemy:\n${issues}` }
        ]
      })
    });

    const fixData = await fixRes.json();
    html = fixData.choices[0].message.content;
  }

  res.json({ finalHTML: html });
});

// ------------------- AUTO SEO -------------------
app.post("/seo", async (req, res) => {
  const { html, keyword } = req.body;
  const prompt = `
  Optymalizuj stronę pod SEO dla frazy: "${keyword}"
  Dodaj <title>, meta description, nagłówki H1-H3, alt do obrazów.
  Zachowaj design, zwróć tylko poprawiony HTML.
  `;

  const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.GROQ_API_KEY}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      model: "llama3-70b-8192",
      messages: [{ role: "user", content: prompt + html }]
    })
  });

  const data = await response.json();
  res.json({ html: data.choices[0].message.content });
});

// ------------------- STRIPE / SUBSKRYPCJE -------------------
app.post("/create-checkout-session", async (req, res) => {
  const { userId, plan } = req.body;

  const session = await stripe.checkout.sessions.create({
    payment_method_types: ['card'],
    line_items: [{ price: plan === 'pro' ? process.env.STRIPE_PRO_PRICE : process.env.STRIPE_FREE_PRICE, quantity: 1 }],
    mode: 'subscription',
    success_url: `${process.env.CLIENT_URL}/dashboard?success=true`,
    cancel_url: `${process.env.CLIENT_URL}/dashboard?canceled=true`,
    metadata: { userId }
  });

  res.json({ id: session.id });
});

// ------------------- SUPABASE USER / LIMITY -------------------
app.post("/use-credit", async (req,res) => {
  const { userId } = req.body;
  const { data: user } = await supabase.from("users").select("*").eq("id", userId).single();

  if(user.plan === "free"){
    if(user.credits <= 0) return res.status(400).json({ error:"Brak kredytów" });
    await supabase.from("users").update({ credits: user.credits - 1 }).eq("id", userId);
  }

  res.json({ success:true });
});

// ------------------- START SERVER -------------------
const PORT = process.env.PORT || 3000;
app.listen(PORT, ()=> console.log(`Server running on port ${PORT}`));
