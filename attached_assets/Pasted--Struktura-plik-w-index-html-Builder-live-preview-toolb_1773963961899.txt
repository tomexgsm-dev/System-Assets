⚙️ builder.js – FULL DRAG & DROP + AI + SEO + TEMPLATES + EXPORT
let aiHTML = "";

// --- DRAG & DROP ---
let dragged;
document.addEventListener("dragstart", e => dragged = e.target);
document.addEventListener("dragover", e => e.preventDefault());
document.addEventListener("drop", e => {
  if(e.target.classList.contains("dropzone")){
    e.target.appendChild(dragged);
    aiHTML = document.getElementById("builder").innerHTML;
    updatePreview();
  }
});

// --- ADD SECTION ---
function addSection(type){
  const builder = document.getElementById("builder");
  let section = document.createElement("section");
  section.setAttribute("draggable", true);

  if(type === 'hero'){
    section.innerHTML = `<h1>Twoja Strona</h1><p>Opis / slogan</p><button>Zamów teraz</button>`;
  }
  if(type === 'menu'){
    section.innerHTML = `<h2>Oferta</h2><div>Basic</div><div>Premium</div><div>VIP</div>`;
  }
  if(type === 'contact'){
    section.innerHTML = `<h2>Kontakt</h2><form><input placeholder="Imię"><input placeholder="Email"><button>Wyślij</button></form>`;
  }

  builder.appendChild(section);
  aiHTML = builder.innerHTML;
  updatePreview();
}

// --- LOAD TEMPLATE ---
function loadTemplate(name){
  fetch(`/templates/${name}.html`).then(r=>r.text()).then(t=>{
    document.getElementById("builder").innerHTML = t;
    aiHTML = t;
    updatePreview();
  });
}

// --- LIVE PREVIEW ---
function updatePreview(){
  document.getElementById("preview").srcdoc = aiHTML;
}

// --- AI AGENT ---
async function runAgent(){
  const res = await fetch("http://localhost:3000/agent", {
    method:"POST",
    headers: {"Content-Type":"application/json"},
    body: JSON.stringify({ html: aiHTML })
  });
  const data = await res.json();
  aiHTML = data.finalHTML;
  updatePreview();
  alert("AI Agent poprawił stronę 🔥");
}

// --- AUTO SEO ---
async function seo(){
  const keyword = document.getElementById("keyword").value;
  const res = await fetch("http://localhost:3000/seo", {
    method:"POST",
    headers: {"Content-Type":"application/json"},
    body: JSON.stringify({ html: aiHTML, keyword })
  });
  const data = await res.json();
  aiHTML = data.html;
  updatePreview();
  alert("SEO zoptymalizowane 🚀");
}

// --- EXPORT HTML ---
function exportHTML(){
  const blob = new Blob([aiHTML], {type:"text/html"});
  const link = document.createElement("a");
  link.download = "strona.html";
  link.href = URL.createObjectURL(blob);
  link.click();
}

// --- EXPORT WORDPRESS (PROSTA KONWERSJA) ---
function exportWordPress(){
  let wpHTML = `<?php /* WordPress Template */ ?>\n` + aiHTML;
  const blob = new Blob([wpHTML], {type:"text/html"});
  const link = document.createElement("a");
  link.download = "wp-strona.php";
  link.href = URL.createObjectURL(blob);
  link.click();
}
