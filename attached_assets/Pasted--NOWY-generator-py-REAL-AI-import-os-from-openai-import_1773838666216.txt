import os
from openai import OpenAI
import requests

# 🔥 OpenAI
client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# 🔥 Claude (Anthropic API przez HTTP)
CLAUDE_API_KEY = os.getenv("CLAUDE_API_KEY")


def generate_with_openai(prompt):
    response = client.chat.completions.create(
        model="gpt-4.1-mini",
        messages=[
            {
                "role": "system",
                "content": "You are a professional web developer. Generate full modern HTML landing page using Tailwind CSS. Return ONLY code."
            },
            {
                "role": "user",
                "content": prompt
            }
        ]
    )

    return response.choices[0].message.content


def generate_with_claude(prompt):
    url = "https://api.anthropic.com/v1/messages"

    headers = {
        "x-api-key": CLAUDE_API_KEY,
        "anthropic-version": "2023-06-01",
        "content-type": "application/json"
    }

    data = {
        "model": "claude-3-haiku-20240307",
        "max_tokens": 2000,
        "messages": [
            {
                "role": "user",
                "content": f"Create modern HTML landing page using Tailwind CSS. Return ONLY code.\n\n{prompt}"
            }
        ]
    }

    res = requests.post(url, headers=headers, json=data)

    return res.json()["content"][0]["text"]


def generate_website(prompt, model="openai"):
    if model == "claude":
        return generate_with_claude(prompt)
    else:
        return generate_with_openai(prompt)

def generate():

async function generate() {
    const prompt = document.getElementById("prompt").value;
    const model = document.getElementById("model").value;

    const res = await fetch("/generate", {
        method: "POST",
        headers: {
            "Content-Type": "application/json"
        },
        body: JSON.stringify({ prompt, model })
    });

    const data = await res.json();

    document.getElementById("preview").srcdoc = data.html;
}
