app.post("/chat", async (req, res) => {
  const { message, html } = req.body;

  const systemPrompt = `
  Jesteś ekspertem frontend developerem i debugerem.
  
  Twoje zadania:
  - analizujesz kod HTML/CSS/JS
  - wykrywasz błędy
  - tłumaczysz problem prostym językiem
  - dajesz konkretne rozwiązania
  
  Odpowiadaj jasno:
  1. Co jest problemem
  2. Dlaczego
  3. Jak to naprawić (kod)
  `;

  const response = await fetch("https://api.groq.com/openai/v1/chat/completions", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${process.env.GROQ_API_KEY}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({
      model: "llama3-70b-8192",
      messages: [
        { role: "system", content: systemPrompt },
        { role: "user", content: `Kod strony:\n${html}\n\nPytanie:\n${message}` }
      ]
    })
  });

  const data = await response.json();
  res.json({ reply: data.choices[0].message.content });
});

async function sendChat() {
